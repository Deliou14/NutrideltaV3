SELECT alim_ssgrp_code, alim_ssgrp_nom_eng, avg(teneur)as teneur_moyenne, const_nom_fr, const_code From compo natural join alim natural join const NATURAL JOIN alim_grp where const_code in ( SELECT const_code FROM const WHERE const_nom_fr REGEXP '(?i)(Calcium|Cuivre|Fer|Iode|Chlorure|Magnesium|Manganese|Potassium|Phosphore|Selenium|Zinc|Vitamine)' ) group by const_code, alim_ssgrp_code ORDER BY teneur_moyenne DESC
